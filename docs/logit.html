inv_logit <- function(x) {
  
  p <- 1 / (1 + exp(-x))
  return(p)
  
}

logit <- function(data, formula, na.action=na.exclude) {
  
  mf <- model.frame(formula, data, na.action=na.action)
  Y <- model.response(mf, type="double")
  X <- model.matrix(mf)
  
  B <- numeric(ncol(X))
  
  logitll <- function(B) {
    
    p <- inv_logit((X %*% B))
    lli <- Y*log(p) + (1-Y)*log(1-p)
    
    return(-sum(lli))
    
  }
  
  est <- optim(B, logitll, method="BFGS")
  
  return(est$par)
  
}
